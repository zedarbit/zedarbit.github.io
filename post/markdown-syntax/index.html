2. 重要子句

1）SELECT

USE sql_store;#大写小写都行，但最好关键字用大写
SELECT *
FROM CUSTOMERS #ABOVE 就是选择customers列的一个标准形式
WHERE customers_id = 1; # 根据条件筛选
ORDER BY first_name # sort
# '#' or '--' would be fine
#子句的顺序会对结果产生影响

# 当想要选择某一列或者是某两列时
SELECT 
	first_name, 
	last_name，
	points， 
	points*0.9 AS "discount num"
# 返回的是人的名字，积分，甚至可以加上数学公式,还可以用AS给一个新的列加一个名字
SELECT DISTINCT state
FROM customers
# 列出客户的州，并且去除重复项

2）WHERE

SELECT *
FROM customers
WHERE points > 3000
-- 遍历每一个用户然后筛出满足的选项
WHERE state = 'VA'
--筛选出符合的元组（大写小写都一样）
WHERE birthday_date > '2000-01-24'
--比较普遍的日期表达方式

3）AND/OR/NOT

SELECT *
FROM customers
WHERE birthday_date > '1990-01-01' OR points > 100 AND state = 'va'
-- 和 && , ||, !  一个作用,但运算符的优先级不一样,所以要加括号： AND > OR 
WHERE NOT (birthday_date > '1990-01-01' OR points > 100)

4）IN

WHERE state = 'VA' OR state = 'GA' OR state  = 'FL'
--筛选出一下几个州的人
WHERE state (NOT) IN ('VA', 'FL', 'GA')
--简化运算

5）BETWEEN 

WHERE points >1000 AND points < 1500
-- equals with
WHERE points BETWEEN 1000 AND 1500

6) LIKE 

WHERE last_name LIKE 'b%'('B%')
--表示摘出以b或者B开头的元素，b反正就是一个字符串 可以简单可以复杂
WHERE last_name LIKE '%b%'
--b在中间
WHERE last_name LIKE '%y'
--以y结尾
WHERE last_name LIKE '_y'
--一共只能有两个字母且最后一个是y，有几个_就代表了几个字母
WHERE last_name LIKE 'b____y'
--b开头 4个_ y结尾

7）REGEXP

which means regular expression

WHERE last_name REGEXP 'field'
--找出含有field的人 == LIKE '%field%'
WHERE last_name REGEXP '^field'
--以field打头 == 'field%'
WHERE last_name REGEXP 'field$'
--以field结尾 == '%field'
WHERE last_name REGEXP 'field|mac|rose'
--含有field或mac或rose
WHERE last_name REGEXP '[gim]e'
WHERE last_name REGEXP '[a-h]e'
--含有ge/ie/me的都会被筛选出来
--直接包含abcdefg和e

正则表达式：

^ begin

$ end

| or

[asd] choose one

[a-z] from to

8）IS NULL

WHERE phone IS (NOT) NULL

9) ORDER BY

ORDER BY first_name 
--默认按照 条件 升序排列
ORDER BY first_name DESC
--降序排列
ORDER BY quantity * unit_price DESC 
--也可以用表达式来排序

10）LIMIT

SELECT * 
FROM customers 
LIMIT 10
--选取前10项
LIMIT 6,3
--跳过前6项，往后取3项
--可用于分页的数目
	--page 1： 1-3
	--page 2： 4-6
	--page 3： 7-9

可以先排序，再取前几个。
